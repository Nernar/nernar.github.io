fromConfig

  • fromConfig(config: Config): void

  • Fetches dimension height ranges from present config, that could be obtained as subvalue via Config.get.

    Learn how to use
    // {
        // "enabled": true,
        // "heights": {
            // "overworld": {
                // height: 512
            // }
        // }
    // }

    InfiniteForest.fromConfig(__config__.get("heights"));

    Parameters

    • config: Config

    Returns void

fromJson

  • fromJson(json: ConfigType): void

  • Fetches dimension height ranges from present object.

    Learn how to use
    InfiniteDepth.fromJson({
        // Could be vanilla overworld, nether or end
        "overworld": {
            // Minimum world level at which bedrock will be located
            "onset": -64,
            // Height limit that is relative to starting coordinate
            "height": 384
        },
        // Could be custom dimension uid, which is used when registration
        "aw_enchanted_forest": {
            // Onset or even minimum coordinate fallbacks to zero
            "height": 320
        },
        // Could be numeric dimension uid, which is always unique
        [InfiniteForest.id]: {
            "min": 0,
            "max": 512
        }
    });

    Parameters

    • json: ConfigType

    Returns void

    Parameters

    • dimension: DimensionType

    Returns number

setDimensionBounds

  • setDimensionBounds(dimension: DimensionType, onset: number, height: number): void

  • Directly sets dimension height range bounds, which will be changed when reentering dimension on server-side.

    Parameters

    • dimension: DimensionType
    • onset: number

      minimum world level at which bedrock will be located, accepts values between -2048 and 2048

    • height: number

      height limit that is relative to starting coordinate, accepts values between 0 and 512

    Returns void

setDimensionHeightRange

  • setDimensionHeightRange(dimension: DimensionType, min: number, max: number): void

  • Directly sets dimension height range coordinates, which will be changed when reentering dimension on server-side.

    Parameters

    • dimension: DimensionType
    • min: number

      minimum world coordinate that cannot be greater than upper level, accepts values between -2048 and 2048

    • max: number

      maximum world coordinate that cannot be less than lower level, accepts values between -2048 and 2048

    Returns void
